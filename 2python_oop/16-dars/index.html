üì¶ Mahsulot sinfi

Birinchi sinf - Mahsulot sinfini yaratamiz:

class Mahsulot:
    def __init__(self, nomi, narxi, miqdor):
        self.nomi = nomi
        self.narxi = narxi
        self.miqdor = miqdor
    
    def malumot(self):
        return f"Mahsulot: {self.nomi}, Narx: {self.narxi} so'm, Miqdor: {self.miqdor}"
    
    def narxni_ozgartir(self, yangi_narx):
        if yangi_narx > 0:
            self.narxi = yangi_narx
            return f"{self.nomi} narxi {yangi_narx} so'mga o'zgartirildi"
        else:
            return "Noto'g'ri narx!"
    
    def miqdorni_ozgartir(self, yangi_miqdor):
        if yangi_miqdor >= 0:
            self.miqdor = yangi_miqdor
            return f"{self.nomi} miqdori {yangi_miqdor}ga o'zgartirildi"
        else:
            return "Noto'g'ri miqdor!"

# Test qilish
telefon = Mahsulot("iPhone 15", 15000000, 5)
print(telefon.malumot())
print(telefon.narxni_ozgartir(16000000))
print(telefon.miqdorni_ozgartir(3))

üë§ Mijoz sinfi

Ikkinchi sinf - Mijoz sinfini yaratamiz:

class Mijoz:
    def __init__(self, ism, telefon, manzil):
        self.ism = ism
        self.telefon = telefon
        self.manzil = manzil
        self.savatcha = []  # Savatcha ro'yxati
        self.jami_summa = 0
    
    def malumot(self):
        return f"Mijoz: {self.ism}, Telefon: {self.telefon}, Manzil: {self.manzil}"
    
    def savatchaga_qoshish(self, mahsulot, miqdor):
        if mahsulot.miqdor >= miqdor:
            # Savatchaga qo'shish
            self.savatcha.append({
                'mahsulot': mahsulot,
                'miqdor': miqdor,
                'summa': mahsulot.narxi * miqdor
            })
            self.jami_summa += mahsulot.narxi * miqdor
            return f"{mahsulot.nomi} savatchaga qo'shildi ({miqdor} dona)"
        else:
            return f"Yetarli mahsulot yo'q! Mavjud: {mahsulot.miqdor}"
    
    def savatchani_korsatish(self):
        if not self.savatcha:
            return "Savatcha bo'sh"
        
        natija = "Savatcha:\n"
        for item in self.savatcha:
            natija += f"- {item['mahsulot'].nomi}: {item['miqdor']} dona, {item['summa']} so'm\n"
        natija += f"Jami: {self.jami_summa} so'm"
        return natija

# Test qilish
mijoz = Mijoz("Ali", "+998901234567", "Toshkent")
print(mijoz.malumot())

üè™ Do'kon sinfi

Uchinchi sinf - Do'kon sinfini yaratamiz:

class Dokon:
    def __init__(self, nomi, manzil):
        self.nomi = nomi
        self.manzil = manzil
        self.mahsulotlar = []  # Mahsulotlar ro'yxati
        self.sotuvlar = []     # Sotuvlar tarixi
    
    def mahsulot_qoshish(self, mahsulot):
        self.mahsulotlar.append(mahsulot)
        return f"{mahsulot.nomi} do'konga qo'shildi"
    
    def mahsulotlar_royxati(self):
        if not self.mahsulotlar:
            return "Do'konda mahsulot yo'q"
        
        natija = f"{self.nomi} mahsulotlari:\n"
        for mahsulot in self.mahsulotlar:
            natija += f"- {mahsulot.malumot()}\n"
        return natija
    
    def mahsulot_topish(self, nomi):
        for mahsulot in self.mahsulotlar:
            if mahsulot.nomi.lower() == nomi.lower():
                return mahsulot
        return None
    
    def sotuv_amalga_oshirish(self, mijoz):
        if not mijoz.savatcha:
            return "Savatcha bo'sh!"
        
        # Sotuvni amalga oshirish
        for item in mijoz.savatcha:
            mahsulot = item['mahsulot']
            miqdor = item['miqdor']
            mahsulot.miqdor -= miqdor
        
        # Sotuvni tarixga qo'shish
        self.sotuvlar.append({
            'mijoz': mijoz.ism,
            'savatcha': mijoz.savatcha.copy(),
            'jami_summa': mijoz.jami_summa
        })
        
        natija = f"Sotuv amalga oshirildi!\n{mijoz.savatchani_korsatish()}"
        
        # Savatchani tozalash
        mijoz.savatcha = []
        mijoz.jami_summa = 0
        
        return natija

# Test qilish
dokon = Dokon("Tech Store", "Toshkent shahar")
print(dokon.mahsulotlar_royxati())

üîó Loyiha birlashtirish

Endi barcha sinflarni birlashtirib, to'liq tizimni yaratamiz:

# Asosiy dastur
def main():
    # Do'kon yaratish
    dokon = Dokon("Tech Store", "Toshkent shahar")
    
    # Mahsulotlar qo'shish
    telefon = Mahsulot("iPhone 15", 15000000, 5)
    noutbuk = Mahsulot("MacBook Pro", 25000000, 3)
    planshet = Mahsulot("iPad", 8000000, 8)
    
    dokon.mahsulot_qoshish(telefon)
    dokon.mahsulot_qoshish(noutbuk)
    dokon.mahsulot_qoshish(planshet)
    
    # Mijoz yaratish
    mijoz = Mijoz("Ali", "+998901234567", "Toshkent")
    
    # Mahsulotlarni ko'rish
    print("=== DO'KON MAHSULOTLARI ===")
    print(dokon.mahsulotlar_royxati())
    
    # Savatchaga qo'shish
    print("\n=== SAVATCHAGA QO'SHISH ===")
    print(mijoz.savatchaga_qoshish(telefon, 2))
    print(mijoz.savatchaga_qoshish(planshet, 1))
    
    # Savatchani ko'rish
    print("\n=== SAVATCHA ===")
    print(mijoz.savatchani_korsatish())
    
    # Sotuv amalga oshirish
    print("\n=== SOTUV ===")
    print(dokon.sotuv_amalga_oshirish(mijoz))
    
    # Yangi holat
    print("\n=== YANGI HOLAT ===")
    print(dokon.mahsulotlar_royxati())

# Dasturni ishga tushirish
if __name__ == "__main__":
    main()
